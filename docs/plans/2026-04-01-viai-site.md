**3. customers.html — Social Proof**
- Testimonials (static JSON, no database)
- Customer logos or names

**4. docs.html — Knowledge Base**
- Search box: Query Dify knowledge base via `/api/rag/search` endpoint
- Categories: Getting Started, Hardware, Troubleshooting, API
- Fallback: If Dify unavailable, show "Offline" message and static content
- No authentication required

### server.py

```python
from fastapi import FastAPI, HTTPException
from fastapi.staticfiles import StaticFiles
from fastapi.responses import FileResponse
import httpx
import os

app = FastAPI()

# Serve static files
app.mount("/static", StaticFiles(directory="static"), name="static")

# ── Health ──
@app.get("/health")
async def health():
    return {"status": "ok"}

# ── API: Fleet Status (demo data or live) ──
@app.get("/api/fleet-status")
async def fleet_status():
    """Return current fleet status for products.html demo."""
    # Option 1: Static demo data
    return {
        "devices": [
            {"id": "Magic-A3F2", "status": "online", "battery": 87, "signal": -42},
            {"id": "Magic-B1E7", "status": "online", "battery": 62, "signal": -58},
            {"id": "Magic-C4D9", "status": "offline", "battery": 0, "signal": -120},
        ],
        "timestamp": "2026-04-01T12:00:00Z"
    }
    # Option 2: Query daemon API (if daemon is running)
    # try:
    #     async with httpx.AsyncClient() as client:
    #         resp = await client.get("http://localhost:8001/api/devices")
    #         return resp.json()
    # except:
    #     return {"error": "Daemon unavailable"}

# ── API: RAG Search (Dify knowledge base) ──
@app.get("/api/rag/search")
async def rag_search(q: str):
    """Search Dify knowledge base."""
    dify_url = os.getenv("DIFY_BASE_URL", "http://localhost:8400/v1")
    dify_key = os.getenv("DIFY_API_KEY")

    if not dify_key:
        raise HTTPException(status_code=503, detail="RAG unavailable")

    try:
        async with httpx.AsyncClient() as client:
            # Call Dify API (simplified; see Dify docs for real endpoint)
            resp = await client.post(
                f"{dify_url}/chat-messages",
                headers={"Authorization": f"Bearer {dify_key}"},
                json={"inputs": {}, "query": q, "user": "viai-site"}
            )
            return resp.json()
    except Exception as e:
        raise HTTPException(status_code=503, detail=f"RAG error: {str(e)}")

# ── Root: Serve index.html ──
@app.get("/")
async def root():
    return FileResponse("index.html")

# Catch-all: Serve index.html for SPA routing
@app.get("/{path:path}")
async def catch_all(path: str):
    return FileResponse("index.html")

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8010)
```

### main.js

Minimal. No frameworks, no state management. Safe DOM manipulation:

```javascript
// Smooth scroll for internal links
document.querySelectorAll('a[href^="#"]').forEach(link => {
  link.addEventListener('click', (e) => {
    e.preventDefault();
    const target = document.querySelector(link.getAttribute('href'));
    if (target) target.scrollIntoView({ behavior: 'smooth' });
  });
});

// Fleet status demo (products.html)
async function loadFleetStatus() {
  try {
    const resp = await fetch('/api/fleet-status');
    const data = await resp.json();
    const container = document.getElementById('fleet-demo');
    if (!container) return;

    // Safe DOM manipulation: create elements, don't use innerHTML with user data
    container.innerHTML = ''; // Clear
    const grid = document.createElement('div');
    grid.className = 'fleet-grid';

    data.devices.forEach(d => {
      const card = document.createElement('div');
      card.className = `device-card ${d.status}`;
      card.textContent = `${d.id} - ${d.status} - Battery: ${d.battery}%`;
      grid.appendChild(card);
    });

    container.appendChild(grid);
  } catch (err) {
    console.error('Fleet load failed:', err);
  }
}

// RAG search (docs.html)
async function searchDocs(query) {
  try {
    const resp = await fetch(`/api/rag/search?q=${encodeURIComponent(query)}`);
    const data = await resp.json();
    // Render results safely (don't use innerHTML for user/API data)
    const resultsDiv = document.getElementById('search-results');
    if (resultsDiv) {
      resultsDiv.textContent = JSON.stringify(data, null, 2);
    }
  } catch (err) {
    console.error('Search failed:', err);
  }
}

// Init
window.addEventListener('DOMContentLoaded', () => {
  if (document.getElementById('fleet-demo')) loadFleetStatus();
});
```

### Style Guide

- **No CSS frameworks** (no Tailwind, no Bootstrap) — plain CSS with variables
- **No JavaScript frameworks** — vanilla DOM manipulation only
- **Responsive breakpoints:** 480px, 768px, 1024px
- **Dark theme only** — no light mode switcher
- **Fast load:** Target <100KB total CSS + JS
- **Safe DOM:** Use textContent for user/API data, never innerHTML with untrusted content
